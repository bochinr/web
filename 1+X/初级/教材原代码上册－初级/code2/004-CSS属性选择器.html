有lang属性的td元素会被选择
th[lang] 无属性 lang="" lang="en" lang="cn"

title属性值为a的td元素会被选择
[title="a"] 无属性 title="a" title="a b" title="ab" title="ba"

title属性值包含“c，且c前后只能有空格”的td元素会被选择
[title~="c"] 无属性 title="c" title="c d" title="c-d" title="cd" title="d c" title="dc"

title属性值为“h开头，且h只能为独立单词，后面可跟连字符”的td元素会被选择
[title|="h"] 无属性 title="h" title="h i" title="h-i" title="hi" title="i h" title="i h j"
